def divtwo(dividend):
    # Gibt dividend / 2 zurück (abgerundet)
    divisor = 0
    if (dividend >= 0):
        while ((divisor + divisor) < (dividend - 1)):
            divisor = (divisor + 1)
    return divisor

def twoToThePowerOf(x):
    # Gibt 2^x zurück
    res = 2
    if (x == 0):
        res = 1
    while (x > 1):
        res = (res + res)
        x = (x - 1)
    return res

def checkIfSeperator(l, i):
    counter = 1
    res = 0
    
    digit = binTestBit(l, i)
    next = binTestBit(l, (i + 1))
    
    if (digit == 1):
        if (next == 0):
            prev = binTestBit(l, (i - 1))
            diff = 2
            while (prev == digit):
                counter = (counter + 1)
                prev = binTestBit(l, (i - diff))
                diff = (diff + 1)
            div = divtwo(counter)
            if (counter > (div + div)):
                res = 1
    return res

def binLength(l):   
    # Wenn l > 0 wird die Länge der Binärzahl zurückegegeben.
    # Sonst: Fehlermeldung, da "counter" nicht definiert ist.
    if (l == 0):
        counter = 1
    if (l > 0):
        counter = 0
        while (l > 0):
            l = divtwo(l)
            counter = (counter + 1)
    return counter

def binTestBit(n, i):
    length = binLength(n)
    i = ((length + 1) - i)
        # Da die Binärziffern von rechts nach links berechnet
        # werden, müssen wir den Index i invertieren.
        # (length + 1), da wir bei 1 anfangen zu zählen
    digit = -1
    if (i < (length + 1)): # Index darf maximal gleich length sein.
        counter = 0
        # Für i < 1 bzw. i > length gibt die Fkt einen Fehler aus, da
        # entweder digit oder counter nicht definiert sind.
        while (counter != i):
            div = divtwo(n)
            digit = (n - (div + div))
            n = div
            counter = (counter + 1)
    return digit

def calcBitValue(digit, power):
    val = 0
    if (digit == 1):
        val = (val + twoToThePowerOf(power))
    return val


def ListGetLength(l):

    for x in range (1, (length + 1)):
